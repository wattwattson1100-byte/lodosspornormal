        <button onclick="addPost()">Paylaş</button>
    </div>
    <div class="feed-container" id="feed"></div>
    <script>
        function addPost() {
            let text = document.getElementById('postText').value;
            let feed = document.getElementById('feed');
            
            let post = document.createElement('div');
            post.className = 'post';
            post.innerHTML = `
                <p>${text}</p>
                <div class="actions">
                    <button onclick="like(this)">❤️ Beğen</button>
                    <input type="text" placeholder="Yorum yaz...">
                </div>
            `;
            feed.prepend(post); // En üste ekler
            document.getElementById('postText').value = '';
        }
        function like(btn) {
            btn.style.color = "red";
            btn.innerText = "❤️ Beğenildi";
        }
    </script>
